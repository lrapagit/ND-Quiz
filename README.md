const LETTERS = ["A","B","C","D"];
const answers = new Array(15).fill(null);
let employeeName = "";
let dateTaken = "";

// ── Start quiz ────────────────────────────────────────────────────────────────
function startQuiz() {
  const first = document.getElementById('first-name').value.trim();
  const last  = document.getElementById('last-name').value.trim();
  const date  = document.getElementById('quiz-date').value;
  let valid = true;

  ['first-name','last-name','quiz-date'].forEach(id => {
    document.getElementById(id).classList.remove('error');
  });
  ['err-first','err-last','err-date'].forEach(id => {
    document.getElementById(id).style.display = 'none';
  });

  if (!first) { document.getElementById('first-name').classList.add('error'); document.getElementById('err-first').style.display='block'; valid=false; }
  if (!last)  { document.getElementById('last-name').classList.add('error');  document.getElementById('err-last').style.display='block';  valid=false; }
  if (!date)  { document.getElementById('quiz-date').classList.add('error');  document.getElementById('err-date').style.display='block';  valid=false; }

  if (!valid) return;

  employeeName = `${first} ${last}`;
  dateTaken = new Date(date + 'T12:00:00').toLocaleDateString('en-US', { year:'numeric', month:'long', day:'numeric' });

  document.getElementById('start-screen').style.display = 'none';
  document.getElementById('quiz-screen').style.display = 'block';
  document.getElementById('progress-wrap').style.display = 'flex';
  renderQuestions();
}

// ── Progress ──────────────────────────────────────────────────────────────────
function updateProgress() {
  const answered = answers.filter(a => a !== null).length;
  document.getElementById('progress-fill').style.width = (answered/15*100)+'%';
  document.getElementById('progress-count').textContent = `${answered} of 15 answered`;
  const btn  = document.getElementById('submit-btn');
  const hint = document.getElementById('submit-hint');
  if (answered === 15) {
    btn.disabled = false;
    hint.textContent = 'All questions answered — ready to submit.';
    hint.style.color = '#2E7D52';
  }
}

// ── Answer a question ─────────────────────────────────────────────────────────
function answer(qIndex, value) {
  if (answers[qIndex] !== null) return;
  answers[qIndex] = value;
  const q = QUESTIONS[qIndex];
  const card = document.getElementById(`card-${qIndex}`);
  const exp  = document.getElementById(`exp-${qIndex}`);
  const isCorrect = value === q.correct;

  card.classList.add(isCorrect ? 'answered-correct' : 'answered-wrong');

  if (q.type === 'mc') {
    for (let i = 0; i < q.choices.length; i++) {
      const btn = document.getElementById(`choice-${qIndex}-${i}`);
      btn.disabled = true;
      if (i === value && isCorrect)  btn.classList.add('selected-correct');
      else if (i === value)          btn.classList.add('selected-wrong');
      else if (i === q.correct)      btn.classList.add('reveal-correct');
    }
  } else {
    ['true','false'].forEach(v => {
      const btn    = document.getElementById(`tf-${qIndex}-${v}`);
      const btnVal = v === 'true';
      btn.disabled = true;
      if (btnVal === value && isCorrect)  btn.classList.add('selected-correct');
      else if (btnVal === value)          btn.classList.add('selected-wrong');
      else if (btnVal === q.correct)      btn.classList.add('reveal-correct');
    });
  }

  exp.style.display = 'block';
  exp.className = `explanation ${isCorrect ? 'correct' : 'wrong'}`;
  let correctAnswerText = '';
  if (!isCorrect) {
    if (q.type === 'mc') {
      correctAnswerText = `<strong>The correct answer is ${LETTERS[q.correct]}: ${q.choices[q.correct]}.</strong> `;
    } else {
      correctAnswerText = `<strong>The correct answer is ${q.correct ? 'True' : 'False'}.</strong> `;
    }
  }
  exp.innerHTML = `<span class="exp-icon">${isCorrect ? '✓ Correct.' : '✗ Incorrect.'}</span>${correctAnswerText}${q.explanation}`;
  updateProgress();
}

// ── Submit quiz ───────────────────────────────────────────────────────────────
async function submitQuiz() {
  const correct = answers.reduce((sum, a, i) => sum + (a === QUESTIONS[i].correct ? 1 : 0), 0);
  const pct     = Math.round(correct / 15 * 100);
  const pass    = correct >= 12;
  const now     = new Date();
  const completedAt = now.toLocaleString('en-US', { year:'numeric', month:'long', day:'numeric', hour:'numeric', minute:'2-digit', timeZoneName:'short' });
  const year    = now.getFullYear();

  document.getElementById('submit-btn').disabled = true;
  const panel = document.getElementById('results-panel');
  panel.style.display = 'block';

  panel.innerHTML = `
    <div class="results-score-circle ${pass?'pass':'fail'}">
      <div class="score-number ${pass?'pass':'fail'}">${correct}</div>
      <div class="score-denom">out of 15</div>
    </div>
    <div class="results-verdict ${pass?'pass':'fail'}">${pass ? 'Quiz Passed' : 'Quiz Not Passed'}</div>
    <div class="results-sub">${pct}% &nbsp;|&nbsp; ${correct} correct, ${15-correct} incorrect</div>

    <div class="results-record">
      <div class="results-record-title">Completion Record</div>
      <div class="record-row"><span class="record-label">Employee Name</span><span class="record-value">${employeeName}</span></div>
      <div class="record-row"><span class="record-label">Date Taken</span><span class="record-value">${dateTaken}</span></div>
      <div class="record-row"><span class="record-label">Submitted At</span><span class="record-value">${completedAt}</span></div>
      <div class="record-row"><span class="record-label">Score</span><span class="record-value">${correct} / 15 &nbsp;(${pct}%)</span></div>
      <div class="record-row"><span class="record-label">Result</span><span class="record-value" style="font-weight:600;color:${pass?'var(--green)':'var(--red)'}">${pass?'PASS':'FAIL'}</span></div>
    </div>

    <div class="sending-status" id="send-status">Sending completion email and logging to tracker&hellip;</div>

    <div class="results-note">
      ${pass
        ? `<strong>Congratulations, ${employeeName}!</strong> You passed the LRAPA Nondiscrimination Program quiz. Your result has been emailed to the Non-Discrimination Coordinator and logged in the annual staff tracking sheet.`
        : `<strong>${employeeName}</strong>, a score of 80% (12/15) is required to pass. You scored ${pct}% (${correct}/15). Please review the documents covering the questions you missed, then retake the quiz. Contact Matt Sorensen at <strong>accessibility@lrapa-or.gov</strong> with any questions.`
      }
    </div>
    <button class="btn-retry" onclick="location.reload()">Retake Quiz</button>
  `;

  panel.scrollIntoView({ behavior:'smooth', block:'start' });

  // ── Send email via EmailJS ────────────────────────────────────────────────
  const templateParams = {
    employee_name: employeeName,
    date_taken:    dateTaken,
    completed_at:  completedAt,
    score:         correct,
    percent:       pct,
    result:        pass ? 'PASS' : 'FAIL',
  };

  // ── Log to Google Sheet via Apps Script ───────────────────────────────────
  const sheetData = {
    timestamp:     completedAt,
    employee_name: employeeName,
    date_taken:    dateTaken,
    score:         correct,
    percent:       pct,
    result:        pass ? 'PASS' : 'FAIL',
    year:          year,
  };

  const statusEl = document.getElementById('send-status');
  let emailOk = false;
  let sheetOk = false;

  try {
    emailjs.init(EMAILJS_PUBLIC_KEY);
    await emailjs.send(EMAILJS_SERVICE_ID, EMAILJS_TEMPLATE_ID, templateParams);
    emailOk = true;
  } catch(e) {
    console.warn('EmailJS error:', e);
  }

  try {
    const resp = await fetch(APPS_SCRIPT_URL, {
      method: 'POST',
      body: JSON.stringify(sheetData),
    });
    const json = await resp.json();
    if (json.status === 'ok') sheetOk = true;
  } catch(e) {
    console.warn('Apps Script error:', e);
  }

  if (emailOk && sheetOk) {
    statusEl.textContent = '✓ Completion email sent and result logged to the annual tracking sheet.';
    statusEl.className = 'sending-status ok';
  } else if (emailOk && !sheetOk) {
    statusEl.textContent = '⚠ Email sent, but the tracking sheet could not be updated. Please contact Matt Sorensen at accessibility@lrapa-or.gov so your result can be logged manually.';
    statusEl.className = 'sending-status err';
  } else if (!emailOk && sheetOk) {
    statusEl.textContent = '⚠ Result logged to tracker, but the completion email could not be sent. Please contact Matt Sorensen at accessibility@lrapa-or.gov to confirm your record.';
    statusEl.className = 'sending-status err';
  } else {
    statusEl.textContent = '⚠ Submission encountered an issue. Please screenshot this entire page and email it to accessibility@lrapa-or.gov so your result can be recorded manually.';
    statusEl.className = 'sending-status err';
  }
}

// ── Render questions ──────────────────────────────────────────────────────────
function renderQuestions() {
  const wrap = document.getElementById('questions-wrap');
  QUESTIONS.forEach((q, i) => {
    const div = document.createElement('div');
    div.className = 'question-card';
    div.id = `card-${i}`;

    let choicesHTML = '';
    if (q.type === 'mc') {
      choicesHTML = `<div class="q-type-badge badge-mc">Multiple Choice</div>
        <div class="choices">` +
        q.choices.map((c, ci) => `
          <button class="choice-btn" id="choice-${i}-${ci}" onclick="answer(${i},${ci})">
            <span class="choice-letter">${LETTERS[ci]}</span><span>${c}</span>
          </button>`).join('') + `</div>`;
    } else {
      choicesHTML = `<div class="q-type-badge badge-tf">True / False</div>
        <div class="tf-choices">
          <button class="tf-btn" id="tf-${i}-true"  onclick="answer(${i},true)">True</button>
          <button class="tf-btn" id="tf-${i}-false" onclick="answer(${i},false)">False</button>
        </div>`;
    }

    div.innerHTML = `
      <div class="q-header">
        <div class="q-number">${q.n}</div>
        <div class="q-text">${q.q}</div>
      </div>
      ${choicesHTML}
      <div class="explanation" id="exp-${i}"></div>`;
    wrap.appendChild(div);
  });
}
